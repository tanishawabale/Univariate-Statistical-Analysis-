mean_charges = data['charges'].mean()
median_charges = data['charges'].median()

print("\nMean of charges:", mean_charges)
print("Median of charges:", median_charges)

if mean_charges > median_charges:
    print("Interpretation: Data is positively skewed (right skewed)")
elif mean_charges < median_charges:
    print("Interpretation: Data is negatively skewed (left skewed)")
else:
    print("Interpretation: Data is normally distributed")
